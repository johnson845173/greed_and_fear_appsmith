select 
concat(first_name,' ',last_name) as name,
product_name,
purchased_at,
phone_number,email,
upi_transaction_id,
transaction_id,
cost
from management.subscription_detail as sd
join users.user_details as ud on ud.user_id = sd.user_id
join management.product_details as pd on pd.product_id = sd.product_id
order by id desc
